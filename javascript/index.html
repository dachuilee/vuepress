# this

谁调用this 就指向谁

let a = 20;
let p = {
  a: 30,
  test:function(){
    alert(this.a);
  }
}
p.test();  // => 30
let s = p.test;
s(); // => 20

q = {
  a: 40,
  function test(){
    function fn(){
      alert(this.a)
    };
    fn();
  }

}
q.test() // fn 没有调用对象 默认为 宿主环境（window）

# 闭包

function f1(){
  var n = 0;
  function f2(){
    n += 1;
    console.log(n);
  }
  return f2;
}
var result = f1(); // 不知道 result 什么时候会被调用 所以会把 函数内的局部变量保存起来
result(); // => 1
result(); // => 2
result = null // 清空内存
